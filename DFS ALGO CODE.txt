graph = {
    'A': ['B', 'C', 'D'],
    'B': [],
    'C': ['E', 'F'],
    'D': [],
    'E': [],
    'F': ['G'],
    'G': []
}

stack = []
path = []

def DFS(graph, start):
    stack.append(start)
    while stack:
        x = stack.pop()
        if x not in path:
            print(x, end=' ')
            path.append(x)
            for node in reversed(graph[x]):
                if node not in path:
                    stack.append(node)

DFS(graph, 'A')
print()
print(path)
