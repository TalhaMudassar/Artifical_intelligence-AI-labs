#bfs code 
import queue
graph = {
    'A':['B','C','D'],
    'B':[],
    'C':['E','F'],
    'D':[],
    'E':[],
    'F':['G'],
    'G':[]
}

path = []
queue = []

def BFS(graph,start):
    queue.append(start)
    path.append(start)

    while queue:
        x=queue.pop(0)
        print(x,end=' ')
        for Node in graph[x]:
            if Node not in path:
                path.append(Node)
                queue.append(Node)

BFS(graph,'A')
print()
